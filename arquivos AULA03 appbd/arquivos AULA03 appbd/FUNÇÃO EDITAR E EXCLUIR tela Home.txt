function excluirItem(id) {

    if (Platform.OS === 'web') {

        const confirmar = window.confirm("Deseja excluir este registro?");

        if (confirmar) {
            deletar(id);
        }

    } else {

        Alert.alert(
            "Excluir",
            "Deseja excluir este registro?",
            [
                { text: "Cancelar" },
                { text: "Excluir", onPress: () => deletar(id) }
            ]
        );

    }

}
